017    package com.google.common.base;
018    
019    import java.nio.charset.Charset;
020    
021    /**
022     * Contains constant definitions for the six standard {@link Charset} instances, which are
023     * guaranteed to be supported by all Java platform implementations.
024     *
025     * @author Mike Bostock
026     * @since 1.0
027     */
028    public final class Charsets {
029      private Charsets() {}
030    
031      /**
032       * US-ASCII: seven-bit ASCII, a.k.a. ISO646-US, a.k.a the Basic Latin block of the Unicode
033       * character set.
034       */
035      public static final Charset US_ASCII = Charset.forName("US-ASCII");
036    
037      /**
038       * ISO-8859-1. ISO Latin Alphabet No. 1, a.k.a. ISO-LATIN-1.
039       */
040      public static final Charset ISO_8859_1 = Charset.forName("ISO-8859-1");
041    
042      /**
043       * UTF-8: eight-bit UCS Transformation Format.
044       */
045      public static final Charset UTF_8 = Charset.forName("UTF-8");
046    
047      /**
048       * UTF-16BE: sixteen-bit UCS Transformation Format, big-endian byte order.
049       */
050      public static final Charset UTF_16BE = Charset.forName("UTF-16BE");
051    
052      /**
053       * UTF-16LE: sixteen-bit UCS Transformation Format, little-endian byte order.
054       */
055      public static final Charset UTF_16LE = Charset.forName("UTF-16LE");
056    
057      /**
058       * UTF-16: sixteen-bit UCS Transformation Format, byte order identified by an optional byte-order
059       * mark.
060       */
061      public static final Charset UTF_16 = Charset.forName("UTF-16");
062    
063      /*
064       * Please do not add new Charset references to this class, unless those character encodings are
065       * part of the set required to be supported by all Java platform implementations! Any Charsets
066       * initialized here may cause unexpected delays when this class is loaded. See the Charset
067       * Javadocs for the list of built-in character encodings.
068       */
069    }
