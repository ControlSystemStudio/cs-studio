017    package com.google.common.base;
018    
019    import java.lang.ref.PhantomReference;
020    import java.lang.ref.ReferenceQueue;
021    
022    /**
023     * Phantom reference with a {@code finalizeReferent()} method which a background thread invokes
024     * after the garbage collector reclaims the referent. This is a simpler alternative to using a
025     * {@link ReferenceQueue}.
026     *
027     * <p>Unlike a normal phantom reference, this reference will be cleared automatically.
028     *
029     * @author Bob Lee
030     * @since 2.0 (imported from Google Collections Library)
031     */
032    public abstract class FinalizablePhantomReference<T> extends PhantomReference<T>
033        implements FinalizableReference {
034      /**
035       * Constructs a new finalizable phantom reference.
036       *
037       * @param referent to phantom reference
038       * @param queue that should finalize the referent
039       */
040      protected FinalizablePhantomReference(T referent, FinalizableReferenceQueue queue) {
041        super(referent, queue.queue);
042        queue.cleanUp();
043      }
044    }
