017    package com.google.common.collect;
018    
019    import com.google.common.annotations.GwtCompatible;
020    
021    import java.util.ListIterator;
022    
023    /**
024     * A list iterator that does not support {@link #remove}, {@link #add}, or
025     * {@link #set}.
026     *
027     * @since 7.0
028     * @author Louis Wasserman
029     */
030    @GwtCompatible
031    public abstract class UnmodifiableListIterator<E>
032        extends UnmodifiableIterator<E> implements ListIterator<E> {
033      /** Constructor for use by subclasses. */
034      protected UnmodifiableListIterator() {}
035    
036      /**
037       * Guaranteed to throw an exception and leave the underlying data unmodified.
038       *
039       * @throws UnsupportedOperationException always
040       */
041      @Override public final void add(E e) {
042        throw new UnsupportedOperationException();
043      }
044    
045      /**
046       * Guaranteed to throw an exception and leave the underlying data unmodified.
047       *
048       * @throws UnsupportedOperationException always
049       */
050      @Override public final void set(E e) {
051        throw new UnsupportedOperationException();
052      }
053    }
