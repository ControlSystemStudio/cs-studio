017    package com.google.common.collect;
018    
019    import com.google.common.annotations.Beta;
020    import com.google.common.annotations.GwtCompatible;
021    
022    import java.util.SortedMap;
023    
024    /**
025     * An object representing the differences between two sorted maps.
026     *
027     * @author Louis Wasserman
028     * @since 8.0
029     */
030    @Beta
031    @GwtCompatible
032    public interface SortedMapDifference<K, V> extends MapDifference<K, V> {
033    
034      @Override
035      SortedMap<K, V> entriesOnlyOnLeft();
036    
037      @Override
038      SortedMap<K, V> entriesOnlyOnRight();
039    
040      @Override
041      SortedMap<K, V> entriesInCommon();
042    
043      @Override
044      SortedMap<K, ValueDifference<V>> entriesDiffering();
045    }
