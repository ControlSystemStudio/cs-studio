017    package com.google.common.collect;
018    
019    import com.google.common.annotations.GwtCompatible;
020    
021    import java.util.Comparator;
022    import java.util.SortedSet;
023    
024    import javax.annotation.Nullable;
025    
026    /**
027     * A sorted set multimap which forwards all its method calls to another sorted
028     * set multimap. Subclasses should override one or more methods to modify the
029     * behavior of the backing multimap as desired per the <a
030     * href="http://en.wikipedia.org/wiki/Decorator_pattern">decorator pattern</a>.
031     *
032     * @author Kurt Alfred Kluever
033     * @since 3.0
034     */
035    @GwtCompatible
036    public abstract class ForwardingSortedSetMultimap<K, V>
037        extends ForwardingSetMultimap<K, V> implements SortedSetMultimap<K, V> {
038    
039      /** Constructor for use by subclasses. */
040      protected ForwardingSortedSetMultimap() {}
041    
042      @Override protected abstract SortedSetMultimap<K, V> delegate();
043    
044      @Override public SortedSet<V> get(@Nullable K key) {
045        return delegate().get(key);
046      }
047    
048      @Override public SortedSet<V> removeAll(@Nullable Object key) {
049        return delegate().removeAll(key);
050      }
051    
052      @Override public SortedSet<V> replaceValues(
053          K key, Iterable<? extends V> values) {
054        return delegate().replaceValues(key, values);
055      }
056    
057      @Override public Comparator<? super V> valueComparator() {
058        return delegate().valueComparator();
059      }
060    }
