017    package com.google.common.annotations;
018    
019    import java.lang.annotation.Documented;
020    import java.lang.annotation.ElementType;
021    import java.lang.annotation.Retention;
022    import java.lang.annotation.RetentionPolicy;
023    import java.lang.annotation.Target;
024    
025    /**
026     * The presence of this annotation on a method indicates that the method may
027     * <em>not</em> be used with the
028     * <a href="http://code.google.com/webtoolkit/">Google Web Toolkit</a> (GWT),
029     * even though its type is annotated as {@link GwtCompatible} and accessible in
030     * GWT.  They can cause GWT compilation errors or simply unexpected exceptions
031     * when used in GWT.
032     *
033     * <p>Note that this annotation should only be applied to methods, fields, or
034     * inner classes of types which are annotated as {@link GwtCompatible}.
035     *
036     * @author Charles Fry
037     */
038    @Retention(RetentionPolicy.CLASS)
039    @Target({ ElementType.TYPE, ElementType.METHOD, ElementType.CONSTRUCTOR, ElementType.FIELD })
040    @Documented
041    @GwtCompatible
042    public @interface GwtIncompatible {
043    
044      /**
045       * Describes why the annotated element is incompatible with GWT. Since this is
046       * generally due to a dependence on a type/method which GWT doesn't support,
047       * it is sufficient to simply reference the unsupported type/method. E.g.
048       * "Class.isInstance".
049       */
050      String value();
051    
052    }
