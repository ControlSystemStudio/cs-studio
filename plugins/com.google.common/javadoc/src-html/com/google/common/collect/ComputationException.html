017    package com.google.common.collect;
018    
019    import com.google.common.annotations.GwtCompatible;
020    
021    /**
022     * Wraps an exception that occurred during a computation.
023     *
024     * @author Bob Lee
025     * @since 2.0 (imported from Google Collections Library)
026     */
027    @GwtCompatible
028    public class ComputationException extends RuntimeException {
029      /**
030       * Creates a new instance with the given cause.
031       */
032      public ComputationException(Throwable cause) {
033        super(cause);
034      }
035      private static final long serialVersionUID = 0;
036    }
