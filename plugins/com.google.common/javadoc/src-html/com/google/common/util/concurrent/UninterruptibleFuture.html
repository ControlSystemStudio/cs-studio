017    package com.google.common.util.concurrent;
018    
019    import com.google.common.annotations.Beta;
020    
021    import java.util.concurrent.ExecutionException;
022    import java.util.concurrent.Future;
023    import java.util.concurrent.TimeUnit;
024    import java.util.concurrent.TimeoutException;
025    
026    /**
027     * A {@code Future} whose {@code get} calls cannot be interrupted. If a thread
028     * is interrupted during such a call, the call continues to block until the
029     * result is available or the timeout elapses, and only then re-interrupts the
030     * thread. Obtain an instance of this type using {@link
031     * Futures#makeUninterruptible(Future)}.
032     *
033     * @author Kevin Bourrillion
034     * @since 1.0
035     * @deprecated Use
036     * {@link Uninterruptibles#getUninterruptibly(Future) getUninterruptibly}.
037     * <b>Scheduled for deletion in Guava Release 11</b>.
038     */
039    @Beta
040    @Deprecated
041    public
042    interface UninterruptibleFuture<V> extends Future<V> {
043      @Override V get() throws ExecutionException;
044    
045      @Override V get(long timeout, TimeUnit unit)
046          throws ExecutionException, TimeoutException;
047    }
