017    package com.google.common.eventbus;
018    
019    import com.google.common.annotations.Beta;
020    import java.lang.annotation.ElementType;
021    import java.lang.annotation.Retention;
022    import java.lang.annotation.RetentionPolicy;
023    import java.lang.annotation.Target;
024    
025    /**
026     * Marks a method as an event handler, as used by
027     * {@link AnnotatedHandlerFinder} and {@link EventBus}.
028     *
029     * <p>The type of event will be indicated by the method's first (and only)
030     * parameter.  If this annotation is applied to methods with zero parameters,
031     * or more than one parameter, the object containing the method will not be able
032     * to register for event delivery from the {@link EventBus}.
033     *
034     * <p>Unless also annotated with @{@link AllowConcurrentEvents}, event handler
035     * methods will be invoked serially by each event bus that they are registered
036     * with.
037     *
038     * @author Cliff Biffle
039     * @since 10.0
040     */
041    @Retention(RetentionPolicy.RUNTIME)
042    @Target(ElementType.METHOD)
043    @Beta
044    public @interface Subscribe {
045    }
